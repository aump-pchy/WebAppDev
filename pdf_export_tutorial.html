เขียน exportPDF function

ใส่ใน <script setup> ของ LogDetailView.vue

LogDetailView.vue — <script setup>
import { ref } from 'vue'
import html2pdf from 'html2pdf.js'

const printArea = ref(null)

function exportPDF() {
  const opt = {
    margin:      [15, 15, 15, 15],  // mm
    filename:    `บันทึกสอน_สัปดาห์${log.value.week}.pdf`,
    image:       { type: 'jpeg', quality: 0.98 },
    html2canvas: {
      scale:       2,          // ความชัด
      useCORS:     true,       // โหลด image จาก Supabase Storage
      windowWidth: 794        // A4 width (px)
    },
    jsPDF: {
      unit:        'mm',
      format:      'a4',
      orientation: 'portrait'
    }
  }

  html2pdf()
    .set(opt)
    .from(printArea.value)
    .save()
}
windowWidth: 794 สำคัญมาก — บอกให้ html2canvas render ที่ความกว้าง A4 พอดี ถ้าไม่ใส่ layout จะเพี้ยน

จัด CSS ให้ตรงแบบฟอร์ม

กำหนด style ของ #print-area ให้ตรงกับ format ต้นฉบับ

<style> — print area
/* กรอบหลัก — กว้าง A4 */
#print-area {
  width:       794px;
  margin:      0 auto;
  padding:     40px;
  font-family: 'Sarabun', sans-serif;
  font-size:   14px;
  line-height: 1.8;
  color:       #000;
  background:  #fff;
}

/* ตาราง */
#print-area table {
  width:           100%;
  border-collapse: collapse;
  font-size:       13px;
  page-break-inside: avoid; /* ห้ามตัดกลางตาราง */
}
#print-area th, #print-area td {
  border:  1px solid #000;
  padding: 4px 8px;
}

/* เส้นใต้แทน input */
.underline-field {
  border-bottom: 1px solid #000;
  min-width:     120px;
  display:       inline-block;
  text-align:    center;
}

/* checkbox style */
.cb-box {
  display:        inline-block;
  width:          12px;
  height:         12px;
  border:         1px solid #000;
  margin-right:   4px;
  vertical-align: middle;
  text-align:     center;
  line-height:    12px;
  font-size:      10px;
}

/* ตัดหน้าตรงที่ต้องการ */
.page-break {
  page-break-before: always;
}
ใช้ .underline-field แทน input เพราะ html2canvas render input ไม่สวย
รูปจาก Supabase Storage ต้องใช้ useCORS: true ถึงจะติดใน PDF

ทดสอบและ troubleshoot

ปัญหาที่พบบ่อยและวิธีแก้

ปัญหาที่พบบ่อย
layout เพี้ยน เพิ่ม windowWidth: 794 และกำหนด width: 794px ใน #print-area
ตารางถูกตัด เพิ่ม page-break-inside: avoid ใน table
รูปไม่ขึ้น ตรวจสอบ useCORS: true และ signed URL ยังไม่หมดอายุ
ภาษาไทยกล่อง โหลด Sarabun font ก่อน export — รอ font load ด้วย document.fonts.ready
รอ font โหลดก่อน export (กรณีภาษาไทยกล่อง)
async function exportPDF() {
  await document.fonts.ready  // รอ Sarabun โหลด
  html2pdf().set(opt).from(printArea.value).save()
}
